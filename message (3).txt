--// Key System GUI (Final Fixed)

local player = game.Players.LocalPlayer
local ScreenGui = Instance.new("ScreenGui")
local Frame = Instance.new("Frame")
local Title = Instance.new("TextLabel")
local KeyBox = Instance.new("TextBox")
local Submit = Instance.new("TextButton")
local InfoText = Instance.new("TextLabel")
local LinkLabel = Instance.new("TextLabel")
local CopyButton = Instance.new("TextButton")

ScreenGui.Parent = player:WaitForChild("PlayerGui")
ScreenGui.ResetOnSpawn = false

-- Frame
Frame.Parent = ScreenGui
Frame.Size = UDim2.new(0,350,0,240)
Frame.Position = UDim2.new(0.5,-175,0.5,-120)
Frame.BackgroundColor3 = Color3.fromRGB(40,40,40)
Frame.Active = true
Frame.Draggable = true

-- Title
Title.Parent = Frame
Title.Size = UDim2.new(1,0,0,40)
Title.BackgroundTransparency = 1
Title.Text = "Key System"
Title.TextColor3 = Color3.new(1,1,1)
Title.TextScaled = true
Title.Font = Enum.Font.SourceSansBold

-- Key Input
KeyBox.Parent = Frame
KeyBox.Size = UDim2.new(0.8,0,0,35)
KeyBox.Position = UDim2.new(0.1,0,0.25,0)
KeyBox.PlaceholderText = "Enter Key..."
KeyBox.TextScaled = true

-- Submit Button
Submit.Parent = Frame
Submit.Size = UDim2.new(0.5,0,0,35)
Submit.Position = UDim2.new(0.25,0,0.43,0)
Submit.Text = "Submit Key"
Submit.TextScaled = true

-- Info Text
InfoText.Parent = Frame
InfoText.Size = UDim2.new(1,0,0,25)
InfoText.Position = UDim2.new(0,0,0.62,0)
InfoText.BackgroundTransparency = 1
InfoText.Text = "Get the key here:"
InfoText.TextColor3 = Color3.new(1,1,1)
InfoText.TextScaled = true

-- NON-EDITABLE LINK
local DISCORD_LINK = "https://discord.gg/AFP7S8bV"

LinkLabel.Parent = Frame
LinkLabel.Size = UDim2.new(0.9,0,0,30)
LinkLabel.Position = UDim2.new(0.05,0,0.75,0)
LinkLabel.BackgroundColor3 = Color3.fromRGB(30,30,30)
LinkLabel.Text = DISCORD_LINK
LinkLabel.TextScaled = true
LinkLabel.TextColor3 = Color3.new(1,1,1)

-- COPY BUTTON
CopyButton.Parent = Frame
CopyButton.Size = UDim2.new(0.4,0,0,28)
CopyButton.Position = UDim2.new(0.3,0,0.88,0)
CopyButton.Text = "Copy Link"
CopyButton.TextScaled = true

-- Copy to Clipboard
CopyButton.MouseButton1Click:Connect(function()
	if setclipboard then
		setclipboard(DISCORD_LINK)
		CopyButton.Text = "Copied!"
		wait(1.5)
		CopyButton.Text = "Copy Link"
	end
end)

-- Key System
local REQUIRED_KEY = "Script"

Submit.MouseButton1Click:Connect(function()
	if KeyBox.Text == REQUIRED_KEY then
		Title.Text = "Access Granted!"
		wait(0.5)
		ScreenGui:Destroy() -- GUI disappears
	else
		Title.Text = "Wrong Key!"
	end
end)
